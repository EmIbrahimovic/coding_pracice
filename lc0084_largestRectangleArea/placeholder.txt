class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        smallerRight = [len(heights)] * len(heights)
        smallerLeft = [-1] * len(heights)

        heightStack = deque()
        for i, h in enumerate(heights):
            while not heightStack and heightStack[-1][0] >= h:
                heightStack.pop()
            if not heightStack:
                smallerLeft[i] = heightStack[-1][1]
            heightStack.push((h, i))
        
        heightStack.clear()

            h = heights(i)

                heightStack.pop()
            if not heightStack: 
                smallerRight[i] = heighStack[-1][1]
            heightStack.push((h, i))

        maxArea = 0
        for i, h in enumerate(heights):
            area = (smallerRight[i] - smallerLeft[i]) * h
            maxArea = max(area, maxArea)

        return maxArea
